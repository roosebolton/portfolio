            .then(res =>
                res.json())

                    handleResponse(response)
                }).catch(error => reject(error))
            })
            .catch(error => reject(error))

        reject(response)
    }
}

        resolve(response)
    } else {
        response.text().then(text => {
            reject(text)
        })
