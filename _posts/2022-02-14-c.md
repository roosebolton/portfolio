website passwordhelper.net aan.

    function randomCaps(words){

    ///creates random integer from inclusive to inclusive
    function getRandomInt(min, max) {       
      // Create byte array and fill with 1 random number
      var byteArray = new Uint8Array(1);
      window.crypto.getRandomValues(byteArray);

      var range = max - min + 1;
      var max_range = 256;
      if (byteArray[0] >= Math.floor(max_range / range) * range)
          return getRandomInt(min, max);
      return min + (byteArray[0] % range);
    }
    
    
    //allCharsNonAmbPW,noNumberPW,allCharNoNumNoAmbPW,allCharPW
    function firstCharSensitive(length, firstArray, firstArrayLength, nonFirstArray, nonFirstArrayLength){
        output = [];
        if(!propperlength(length)){
          return "Password must be at least 8 characters long";
        }
        else{
          rand1= getRandomInt(0, firstArrayLength-1);
          output[0] = firstArray[rand1];
          for(let i = 1; i < length; i++){
            rand2 = getRandomInt(0, nonFirstArrayLength);
            output[i] = nonFirstArray[rand2];
        }
          return output.join("");
        }
    }
    
    //nonCharNonNumNonAmbPW,nonCharNonAmbPW,nonCharNonNumPW,nonCharPW
    function firstCharUnsensitive(length, array, arrayLength){
        output = [];
        if(!propperlength(length)){
          return "Password must be at least 8 characters long";
        }
        else{
          for(let i = 0; i < length; i++){
            rand = getRandomInt(0, arrayLength-1);
            output[i] = array[rand];
        }
          return output.join("");
        }
    }

        }
        ///////all characters
        if((noSpec==false && noNum==false) && noAmb==false){
            return firstCharSensitive(pwLngth,acf,acf.length,acnf,acnf.length);
        }
        ///////no special characters, no ambiguous characters, includes numbers
        if((noSpec==true && noNum==false) && noAmb==true){
            return firstCharUnsensitive(pwLngth,nsnac,nsnac.length);
        }
        ///////no special characters
        if((noSpec==true && noNum==false) && noAmb==false){
            return firstCharUnsensitive(pwLngth,nsc,nsc.length);
